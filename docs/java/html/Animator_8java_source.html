25 /**
26  * Managing timed events
27  */
28 public class Animator {
29  public static abstract class Event {
30  /**
31  * Performs an event
32  * @return the time in milliseconds in which the event must be invoked again; a negative value means "never"
33  */
34  public abstract long run();
35  }

37  private static final Object lock = new Object(); //!< main access control
38  private static Animator instance = null; //!< the instance
39 
40  private boolean threadRunning; //!< while `true`, the thread is running
41  private final Thread thread; //!< the animator thread
42  private Map<Integer, Event> events; //!< list of planned events, indexed by ticket numbers
43  private Map<Integer, Long> eventTimes; //!< list of time points when the planned events are, indexed by ticket numbers
44  private int ticketCounter; //!< counter issuing new ticket numbers
45  private long nextWakeUpTime; //!< time point when the thread is waking up
46 
47  public static final int INVALID_TICKET = -123456789;
48  private static final long TIME_TOLERANCE = 3;
49 
50  private Animator() {
51  events = new HashMap<>();
52  eventTimes = new HashMap<>();
53  threadRunning = true;
54  ticketCounter = 0;
55  nextWakeUpTime = 0;
56  thread = new Thread(new Runnable() {
57  @Override
58  public void run() {
59  while (threadRunning) {
60  try {
61  threadBody();
62  } catch (InterruptedException e) {
63  e.printStackTrace();
64  }
65  }
66  }
67  }, "Beatmup display animator" );
68  thread.start();
69  }

72  private void threadBody() throws InterruptedException {
73  // if there is no events, just sleep
74  if (events.isEmpty())
75  synchronized (thread) {
76  thread.wait();
77  }
78 
79  // pick an event first
80  Event toDoNow = null;
81  int id = INVALID_TICKET;
82  final long now = getCurrentTime();
83  long earliestEvent = Long.MAX_VALUE;
84  synchronized (lock) {
85  for (Map.Entry<Integer, Long> entry : eventTimes.entrySet()) {
86  long time = entry.getValue();
87  earliestEvent = Math.min(earliestEvent, time);
88  if (time <= now + TIME_TOLERANCE) {
89  id = entry.getKey();
90  toDoNow = events.get(id);
91  break;
92  }
93  }
94  }
95 
96  // no event to do right now, sleep and quit
97  if (toDoNow == null) {
98  nextWakeUpTime = earliestEvent;
99  long delay = nextWakeUpTime - getCurrentTime();
100  if (delay > 0)
101  synchronized (thread) {
102  thread.wait(delay);
103  }
104  return;
105  }
106 
107  // there is an event, run it
108  long next = toDoNow.run();
109  while (0 <= next && next <= TIME_TOLERANCE)
110  next = toDoNow.run();
111  synchronized (lock) {
112  if (next > 0)
113  eventTimes.put(id, getCurrentTime()+next);
114  else {
115  events.remove(id);
116  eventTimes.remove(id);
117  }
118  }
119  }
120

123  return System.currentTimeMillis();
124  }

140  /**
141  * Pushes an event to the queue, assigning a unique ticket
142  * @param event the event
143  * @param time time in milliseconds to invoke the event
144  * @return the ticket number
145  */
146  public int planify(Event event, int time) {
147  final long now = getCurrentTime();
148  if (time >= 0)
149  synchronized (lock) {
150  // adding new event
151  long plannedTime = now + time;
152  events.put(ticketCounter, event);
153  eventTimes.put(ticketCounter, plannedTime);
154  // if the thread sleeps at the planned time point, wake it up now
155  if (plannedTime < nextWakeUpTime || nextWakeUpTime < now)
156  synchronized (thread) {
157  thread.notify();
158  }
159  return ticketCounter++;
160  }
161  return INVALID_TICKET;
162  }
163 
164  /**
165  * Cancels a planified event
166  * @param event the event to cancel
167  * @param ticket its ticket number
168  * @return `true` if the event is cancelled successfully
169  */
170  public boolean cancel(Event event, int ticket) {
171  if (ticket != INVALID_TICKET && event != null)
172  synchronized (lock) {
173  if (events.remove(ticket) == event) {
174  eventTimes.remove(ticket);
175  return true;
176  }
177  }
178  return false;
179  }
180 }
