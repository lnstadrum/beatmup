21 import Beatmup.*;

23 /**
24  * Conditional multiple tasks execution.
25  *
26  * Beatmup offers a number of tools allowing to pipeline several tasks into a single one. This technique is particularly useful for designing complex multi-stage image processing
27  * pipelines.
28  *
29  * Multitask is the simplest such tool. It allows to concatenate different tasks into a linear conveyor and run them all or selectively. To handle this selection, each task is
30  * associated with a repetition policy specifying the conditions whether this given task is executed or ignored when the pipeline is running.
31  *
32  * Specifically, there are two extreme modes that force the task execution every time (REPEAT_ALWAYS) or its unconditional skipping (IGNORE_ALWAYS) and two more sophisticated modes
33  * with the following behavior:
34  * - IGNORE_IF_UPTODATE skips the task if no tasks were executed among the ones coming before the current task in the pipeline;
35  * - REPEAT_UPDATE forces task repetition one time on next run and just after switches the repetition policy to IGNORE_IF_UPTODATE.
36  */
37 public class Multitask extends CustomPipeline {
38  private static native long newMultitask(Context context);
39  private native int getRepetitionPolicy(long handle, long taskHolderHandle);
40  private native void setRepetitionPolicy(long handle, long taskHolderHandle, int policy);
41 
42  /**
43  * Specification of conditions whether a specified task in the sequence should be executed
44  */
45  public enum RepetitionPolicy {
46  REPEAT_ALWAYS, //!< execute the task unconditionally on each run
47  REPEAT_UPDATE, //!< execute the task one time then switch to IGNORE_IF_UPTODATE
48  IGNORE_IF_UPTODATE, //!< do not execute the task if no preceding tasks are run
49  IGNORE_ALWAYS //!< do not execute the task
50  }
51 
52  /**
53  * Creates a new instance of Multitask
54  * @param context A Beatmup context
55  */
56  public Multitask(Context context) {
57  super(context, newMultitask(context));
58  }
59 
60  /**
61  * Puts a new task into the end of the task list of the pipeline
62  * @param task the new task
63  * @param policy the new task repetition policy
64  * @return task holder wrapping the new task
65  */
66  public TaskHolder addTask(Task task, RepetitionPolicy policy) {
67  TaskHolder holder = addTask(task);
68  setRepetitionPolicy(handle, holder.handle, policy.ordinal());
69  return holder;
70  }
71 
72  /**
73  * Retrieves repetition policy of a specified task
74  * @param taskHolder the task holder
75  * @return the task repetition policy
76  */
77  public RepetitionPolicy getRepetitionPolicy(TaskHolder taskHolder) {
78  return RepetitionPolicy.values()[ getRepetitionPolicy(handle, taskHolder.handle) ];
79  }

84  * @param policy The new policy
85  */
86  public void setRepetitionPolicy(TaskHolder taskHolder, RepetitionPolicy policy) {
87  setRepetitionPolicy(handle, taskHolder.handle, policy.ordinal());
88  }
