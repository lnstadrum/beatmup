362  AffineMapping newbie = new AffineMapping();
363  newbie.x = x + a11*mapping.x + a12*mapping.y;
364  newbie.y = y + a21*mapping.x + a22*mapping.y;
365  newbie.a11 = a11*mapping.a11 + a12*mapping.a21;
366  newbie.a12 = a11*mapping.a12 + a12*mapping.a22;

372  /**
373  * Computes mappings composition `L` * `R` and assigns it to `target`.
374  * @param target a mapping to assign the composition to
375  * @param L left term of the composition
376  * @param R right term of the composition

378  public static void compose(AffineMapping target, AffineMapping L, AffineMapping R) {
379  float _x = L.x + L.a11*R.x + L.a12*R.y;
380  target.y = L.y + L.a21*R.x + L.a22*R.y;
381  target.x = _x;
382  float
383  _a11 = L.a11*R.a11 + L.a12*R.a21,
384  _a12 = L.a11*R.a12 + L.a12*R.a22,
385  _a21 = L.a21*R.a11 + L.a22*R.a21;
386  target.a22 = L.a21*R.a12 + L.a22*R.a22;
387  target.a11 = _a11;
388  target.a12 = _a12;
389  target.a21 = _a21;

425  public boolean getInverse(AffineMapping inverse) {
426  float det = a11*a22 - a12*a21;
427  if (det == 0)
428  return false;
429  float
430  _a11 = a22 / det,
431  _a22 = a11 / det,
432  _a12 = -a12 / det,
433  _a21 = -a21 / det;
434  inverse.a11 = _a11;
435  inverse.a12 = _a12;
436  inverse.a21 = _a21;
437  inverse.a22 = _a22;

458  /**
459  * Resolves an orthogonal mapping that transforms segment (X1,Y1),(X2,Y2) into another segment.
460  * @param X1 first input point horizontal coordinate
461  * @param Y1 first input point vertical coordinate
462  * @param X2 second input point horizontal coordinate
463  * @param Y2 second input point vertical coordinate
464  * @param mappedX1 first mapped point horizontal coordinate
465  * @param mappedY1 first mapped point vertical coordinate
466  * @param mappedX2 second mapped point horizontal coordinate
467  * @param mappedY2 second mapped point vertical coordinate
468  * @param minScale lower limit on mapping scale
469  * @param maxScale upper limit on mapping scale
470  * @return the mapping itself after resolving
471  */
472  public AffineMapping resolve(
473  float X1, float Y1, float X2, float Y2,
474  float mappedX1, float mappedY1, float mappedX2, float mappedY2,
475  float minScale, float maxScale
476  ) {
477  float
478  x1 = X2 - X1,
479  y1 = Y2 - Y1,
480  n = x1*x1 + y1*y1,
481  x2 = (mappedX2 - mappedX1) / n,
482  y2 = (mappedY2 - mappedY1) / n;
483  a11 = a22 = x1 * x2 + y1 * y2;
484  a12 = y1 * x2 - x1 * y2;
485  a21 = -a12;
486  // respect scale limits
487  float scale2 = a11*a22 - a12*a21;
488  if (scale2 < minScale * minScale) {
489  float f = minScale / (float) Math.sqrt(scale2);
490  a11 *= f;
491  a12 *= f;
492  a21 *= f;
493  a22 *= f;
494  } else
495  if (scale2 > maxScale * maxScale) {
496  float f = maxScale / (float) Math.sqrt(scale2);
497  a11 *= f;
498  a12 *= f;
499  a21 *= f;
500  a22 *= f;
501  }
502  float cX = (X1+X2)/2, cY = (Y1+Y2)/2;
503  x = (mappedX1+mappedX2)/2 - (a11 * cX + a12 * cY);
504  y = (mappedY1+mappedY2)/2 - (a21 * cX + a22 * cY);
505  return this;
506  }
