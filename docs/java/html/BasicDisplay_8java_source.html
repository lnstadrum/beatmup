73  /**
74  * Binds the display as current in the renderer.
75  * The renderer output is then shown on the display (after the next rendering session).
76  */
77  private void makeCurrent() {
78  Surface surface = getHolder().getSurface();
79  if (renderer != null && surface.isValid()) {
80  for (OnBindingListener listener : bindingListeners)
81  listener.beforeBinding(true);
82 
83  try {
84  if (!bindSurfaceToContext(renderer.getContext(), surface))
85  Log.e("BasicDisplay", "Unable to bind the surface");
86  } catch (CoreException e) {
87  e.printStackTrace();
88  }
89 
90  for (OnBindingListener listener : bindingListeners)
91  listener.afterBinding(true);
92  }
93  }
94 
95  /**
96  * Unbinds the display as current from renderer.
97  */
98  private void unmakeCurrent() {
99  if (renderer != null) {
100  for (OnBindingListener listener : bindingListeners)
101  listener.beforeBinding(false);
102 
103  try {
104  if (!bindSurfaceToContext(renderer.getContext(), null))
105  Log.e("BasicDisplay", "Unable to unbind the surface");
106  } catch (CoreException e) {
107  e.printStackTrace();
108  }
109 
110  for (OnBindingListener listener : bindingListeners)
111  listener.afterBinding(false);
112  }
113  }

115  public BasicDisplay(final android.content.Context context, AttributeSet attrs) {
