157  /**
158  * Searches for a camera with a specified facing and selects it if found.
159  * @param facing The wanted facing
160  * @return `true` if found and selected, `false` otherwise (the selected camera remains unchanged then).
161  */
162  public boolean selectCamera(Facing facing) {
163  CameraCharacteristics chars;
164  for (int i = 0; i < cameraIds.length; i++) {
165  try {
166  chars = manager.getCameraCharacteristics(cameraIds[i]);
167  } catch (CameraAccessException e) {
168  if (callback == null)
169  e.printStackTrace();
170  else
171  callback.onAccessError(this);
172  return false;
173  }
174 
175  Integer facingGot = chars.get(CameraCharacteristics.LENS_FACING);
176  if (facingGot != null)
177  switch (facing) {
178  case FRONT:
179  if (facingGot == CameraCharacteristics.LENS_FACING_FRONT) {
180  selectCamera(i);
181  return true;
182  }
183  break;
184 
185  case BACK:
186  if (facingGot == CameraCharacteristics.LENS_FACING_BACK) {
187  selectCamera(i);
188  return true;
189  }
190  break;
191 
192  case EXTERNAL:
193  if (facingGot == CameraCharacteristics.LENS_FACING_EXTERNAL) {
194  selectCamera(i);
195  return true;
196  }
197  break;
198  }
199  }
200 
201  // not found
202  return false;
203  }

212  /**
213  * Performs automatic search of the preview resolution according to a specified criterion.
214  * @param width Desired preview width
215  * @param height Desired preview height
216  * @param policy Criterion used to choose the resolution
217  * @return selected resolution.
218  */
219  public Size chooseResolution(int width, int height, ResolutionSelectionPolicy policy) {
220  if (!checkAccess())
221  return null;
222 
223  StreamConfigurationMap map = characteristics.get(CameraCharacteristics.SCALER_STREAM_CONFIGURATION_MAP);
224  assert map != null;
225  Size[] sizes = map.getOutputSizes(SurfaceTexture.class);
226 
227  switch (policy) {
228  case SMALLEST_COVERAGE:
229  for (Size size : sizes) {
230  if (resolution == null || (
231  size.getWidth() >= width &&
232  size.getHeight() >= height &&
233  size.getWidth() * size.getHeight() < resolution.getWidth() * resolution.getHeight()
234  ))
235  resolution = size;
236  }
237  break;
238 
239  case BIGGEST:
240  for (Size size : sizes) {
241  if (resolution == null ||
242  (size.getWidth() * size.getHeight() >= resolution.getWidth() * resolution.getHeight()))
243  resolution = size;
244  }
245  break;
246  }
247 
248  return resolution;
249  }

258  /**
259  * Retrieves camera image orientation with respect to a given display.
260  * @param display The display
261  * @return counterclockwise camera image orientation in degrees within [0..360) range.
262  */
263  public int getOrientation(Display display) {
264  if (!checkAccess())
265  return 0;
266 
267  // get display orientation
268  int displayOrientation = display.getRotation();
269  if (displayOrientation == android.view.OrientationEventListener.ORIENTATION_UNKNOWN)
270  return 0;
271  displayOrientation = (displayOrientation + 45) / 90 * 90;
272 
273  // Reverse device orientation for front-facing cameras
274  Integer facing = characteristics.get(CameraCharacteristics.LENS_FACING);
275  boolean facingFront = facing != null && facing == CameraCharacteristics.LENS_FACING_FRONT;
276  if (facingFront)
277  displayOrientation = -displayOrientation;
278 
279  Integer sensorOrientation = characteristics.get(CameraCharacteristics.SENSOR_ORIENTATION);
280  if (sensorOrientation == null)
281  sensorOrientation = 0;
282  return (sensorOrientation + displayOrientation + 360) % 360;
283  }

285  /**
286  * Enables face detection, if available.
287  * @return `true` on success.
288  */
289  public boolean enableFaceDetection() {
290  if (captureRequestBuilder == null)
291  return false;
292 
293  int modes[] = characteristics.get(CameraCharacteristics.STATISTICS_INFO_AVAILABLE_FACE_DETECT_MODES);
294  if (modes == null || modes.length == 0 ||
295  (modes.length == 1 && modes[0] == CaptureRequest.STATISTICS_FACE_DETECT_MODE_OFF))
296  return false;
297  captureRequestBuilder.set(CaptureRequest.STATISTICS_FACE_DETECT_MODE, modes[modes.length - 1]);
298 
299  return true;
300  }

363  @Override
364  public void onOpened(CameraDevice cameraDevice) {
365  this.device = cameraDevice;
366  SurfaceTexture texture = image.getSurfaceTexture();
367  assert texture != null;
368  texture.setDefaultBufferSize(resolution.getWidth(), resolution.getHeight());
369  Surface surface = new Surface(texture);
370  try {
371  captureRequestBuilder = cameraDevice.createCaptureRequest(CameraDevice.TEMPLATE_PREVIEW);
372  } catch (CameraAccessException e) {
373  if (callback == null)
374  e.printStackTrace();
375  else
376  callback.onAccessError(Camera.this);
377  return;
378  }
379  captureRequestBuilder.addTarget(surface);
380  try {
381  cameraDevice.createCaptureSession(Arrays.asList(surface), stateCallback, backgroundHandler);
382  } catch (CameraAccessException e) {
383  if (callback == null)
384  e.printStackTrace();
385  else
386  callback.onAccessError(Camera.this);
387  return;
388  }
389 
390  if (callback != null)
391  callback.onCameraOpened(Camera.this);
392  }
