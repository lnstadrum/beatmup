26 /**
27  * A binary operation on images.
28  * Evaluates expression O = op(L, R) for bitmaps L, R, O and a given pixelwise operation op.
29  * Allows to select the operating area in all the three bitmaps.
30  */
31 public class BinaryOperation extends Task {

46  private Bitmap leftOperand, rightOperand, output;
47  private Operation operation;

49  /**
50  * Binary operation specification.
51  */
52  public enum Operation {

130  /**
131  * Specifies a rectangular operating area.
132  * Only pixels within a specific rectangle in left and right operands are used. The result is put in a rectangular area of the same size in the output bitmap (the remaining
133  * pixels are unchanged).
134  * @param width Width in pixels
135  * @param height Height in pixels.
136  * @param leftOrigin Position of the operating area in the left operand.
137  * @param rightOrigin Position of the operating area in the right operand.
138  * @param outputOrigin Position of the operating area in the output image.
139  */
140  public void setCrop(int width, int height, IntPoint leftOrigin, IntPoint rightOrigin, IntPoint outputOrigin) {
141  setCrop(handle,

144  rightOrigin.x, rightOrigin.y,
145  outputOrigin.x, outputOrigin.y);
146  }
147 
148  /**

150  * If the input images are not of the same size as the output, an exception is thrown during the task execution.
151  */
152  public void resetCrop() {
153  resetCrop(handle);
154  }
