61  /**
62  * Sets a predefined color matrix from a preset.
63  * @param preset the preset to set
64  */
65  public void setPreset(Preset preset) throws IllegalArgumentException {
66  try {
67  switch (preset) {
68  case IDENTITY:
69  setCoefficients(0, 0, 1, 0, 0, 0);
70  setCoefficients(1, 0, 0, 1, 0, 0);
71  setCoefficients(2, 0, 0, 0, 1, 0);
72  setCoefficients(3, 0, 0, 0, 0, 1);
73  return;
74  case GRAYSCALE:
75  setCoefficients(0, 0, 1.0f / 3, 1.0f / 3, 1.0f / 3, 0);
76  setCoefficients(1, 0, 1.0f / 3, 1.0f / 3, 1.0f / 3, 0);
77  setCoefficients(2, 0, 1.0f / 3, 1.0f / 3, 1.0f / 3, 0);
78  setCoefficients(3, 0, 0, 0, 0, 1);
79  return;
80  case NEGATIVE:
81  setCoefficients(0, 1, -1, 0, 0, 0);
82  setCoefficients(1, 1, 0, -1, 0, 0);
83  setCoefficients(2, 1, 0, 0, -1, 0);
84  setCoefficients(3, 1, 0, 0, 0, 1);
85  return;
86  }
87  } catch (CoreException whatever) {}
88  throw new IllegalArgumentException("Unknown preset: " + preset.toString());
89  }

101  /**

104  * @param saturationFactor Saturation scaling factor (in HSV sense)
105  * @param valueFactor Value scaling factor (in HSV sense)
106  */
107  public void setColorInversion(Color preservedColor, float saturationFactor, float valueFactor) {

109  }
