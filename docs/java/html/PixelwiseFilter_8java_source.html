19 package Beatmup.Imaging.Filters;

21 import Beatmup.Bitmap;
22 import Beatmup.Context;
23 import Beatmup.Task;

25 /**
26  * Base class for image filters processing a given bitmap in a pixelwise fashion.
27  */
28 class PixelwiseFilter extends Task {
29  protected PixelwiseFilter(Context context, long handle) {
30  super(context, handle);
31  }
32 
33  private native void setInput(long handle, Bitmap input);
34  private native void setOutput(long handle, Bitmap output);
35 
36  private Bitmap input, output;
37 
38  /**
39  * Sets an image to process.
40  * @param input The image
41  */
42  public void setInput(Bitmap input) {
43  this.input = input;
44  setInput(handle, input);
45  }
46 
47  /**
48  * Sets an image receiving the filter result.
49  * If the output image is not of the same size as the input image, an exception is thrown during the task execution.
50  * @param output The image
51  */
52  public void setOutput(Bitmap output) {
53  this.output = output;
54  setOutput(handle, output);
55  }
56 
57  /**
58  * @return the input image or null if not set.
59  */
60  public Bitmap getInput() {
61  return input;
62  }
63 
64  /**
65  * @return the output image or null if not set.
66  */
67  public Bitmap getOutput() {
68  return output;
69  }
70 }
