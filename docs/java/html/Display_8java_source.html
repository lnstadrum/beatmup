53  private class TouchListener implements OnTouchListener {
54  static final int INVALID_POINTER_ID = -1;
55 
56  PointF

58  secondStartPos, //!< initial position of the second pointer
59  mainLastPos,

68  public TouchListener() {

80  private void updateGestureMapping() {
81  SceneRenderer renderer = Display.this.getRenderer();
82  if (renderer != null)
83  switch (renderer.getOutputMapping()) {
84  case FIT_WIDTH_TO_TOP:
85  scaleX = scaleY = 1.0f / getWidth();
86  shiftX = shiftY = 0;
87  return;
88  case FIT_WIDTH:
89  scaleX = scaleY = 1.0f / getWidth();
90  shiftX = 0;
91  shiftY = (getWidth() - getHeight()) / 2;
92  return;
93  case FIT_HEIGHT:
94  scaleX = scaleY = 1.0f / getHeight();
95  shiftX = (getHeight() - getWidth()) / 2;
96  shiftY = 0;
97  return;
98  }
99  scaleX = scaleY = 1.0f / getWidth();
100  shiftX = shiftY = 0;
101  }

104  @Override
105  public boolean onTouch(View v, MotionEvent event) {
106  if (!userActionsEnabled || gestureListener == null)
107  return true;
108 
109  switch (event.getActionMasked()) {
110  // main pointer down: new gesture
111  case MotionEvent.ACTION_DOWN: {
112  updateGestureMapping();
113  final int ptrIdx = event.getActionIndex();
114  if (ptrIdx < 0)
115  break;
116  mainStartPos.set(
117  scaleX * (event.getX(ptrIdx) + shiftX),
118  scaleY * (event.getY(ptrIdx) + shiftY)
119  );
120  mainLastPos.set(mainStartPos.x, mainStartPos.y);
121  mainPtrId = event.getPointerId(ptrIdx);
122  gestureListener.pointerDown(mainStartPos.x, mainStartPos.y);
123  break;
124  }
125 
126  // additional pointer down
127  case MotionEvent.ACTION_POINTER_DOWN: {
128  final int ptrIdx = event.getActionIndex();
129  if (ptrIdx < 0)
130  break;
131  float x = scaleX * (event.getX(ptrIdx) + shiftX);
132  float y = scaleY * (event.getY(ptrIdx) + shiftY);
133  if (secondPtrId == INVALID_POINTER_ID) {
134  secondPtrId = event.getPointerId(ptrIdx);
135  secondStartPos.set(x, y);
136  secondLastPos.set(secondStartPos.x, secondStartPos.y);
137  mainStartPos.set(mainLastPos.x, mainLastPos.y);
138  gestureListener.cumulateGesture();
139  }
140  gestureListener.pointerDown(x, y);
141  break;
142  }
143 
144  case MotionEvent.ACTION_MOVE: {
145  final int mainIdx = event.findPointerIndex(mainPtrId);
146  if (mainIdx < 0) {
147  // invalidation of the main pointer
148  mainPtrId = INVALID_POINTER_ID;
149  break;
150  }
151 
152  mainLastPos.set(
153  scaleX * (event.getX(mainIdx) + shiftX),
154  scaleY * (event.getY(mainIdx) + shiftY)
155  );
156 
157  // if both pointers got, rotation/scaling
158  if (secondPtrId != INVALID_POINTER_ID) {
159  final int secondIdx = event.findPointerIndex(secondPtrId);
160  if (secondIdx >= 0) {
161  secondLastPos.set(
162  scaleX * (event.getX(secondIdx) + shiftX),
163  scaleY * (event.getY(secondIdx) + shiftY)
164  );
165  gestureListener.gestureUpdated(
166  mainStartPos.x, mainStartPos.y, secondStartPos.x, secondStartPos.y,
167  mainLastPos.x, mainLastPos.y, secondLastPos.x, secondLastPos.y
168  );
169  }
170  }
171  // otherwise just translation
172  else
173  gestureListener.gestureUpdated(
174  mainStartPos.x, mainStartPos.y,
175  Float.NaN, Float.NaN,
176  mainLastPos.x, mainLastPos.y,
177  Float.NaN, Float.NaN
178  );
179  break;
180  }
181 
182  case MotionEvent.ACTION_UP:
183  mainPtrId = INVALID_POINTER_ID;
184  secondPtrId = INVALID_POINTER_ID;
185  gestureListener.pointerUp();
186  gestureListener.cancelGesture(); // sometimes some pointers remain down
187  break;
188 
189  case MotionEvent.ACTION_CANCEL:
190  mainPtrId = INVALID_POINTER_ID;
191  secondPtrId = INVALID_POINTER_ID;
192  gestureListener.cancelGesture();
193  break;
194 
195  case MotionEvent.ACTION_POINTER_UP: {
196  final int
197  ptrIdx = event.getActionIndex(),
198  pointerId = event.getPointerId(ptrIdx);
199  if (pointerId == mainPtrId && secondPtrId != INVALID_POINTER_ID) {
200  mainPtrId = secondPtrId;
201  mainLastPos.set(secondLastPos.x, secondLastPos.y);
202  }
203  mainStartPos.set(mainLastPos.x, mainLastPos.y);
204  secondPtrId = INVALID_POINTER_ID;
205  gestureListener.pointerUp();
206  gestureListener.cumulateGesture();
207  break;
208  }
209  }
210 
211  return true;
212  }
213  }
