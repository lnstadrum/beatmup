26 /**
27  * Task rendering a Scene.
28  * The rendering may be done to a given bitmap or on screen (default).
29  * When rendering on screen, the image is sent to BasicDisplay bound to the Context running the rendering task.
30  */
31 public class SceneRenderer extends Task {
32  // references to protect things of being garbage-collected while they are actually used in the native code
33  private Scene scene;
34  private Bitmap background;
35  private Bitmap output;

38  private static native long newSceneRenderer(Context ctx);
39  private native void setOutput(long handle, Bitmap bitmap);
40  private native void resetOutput(long handle);
41  private native void setScene(long handle, Scene scene);
42  private native void setOutputMapping(long handle, int mapping);
43  private native int getOutputMapping(long handle);
44  private native void setOutputReferenceWidth(long handle, int width);
45  private native int getOutputReferenceWidth(long handle);
46  private native void setOutputPixelsFetching(long handle, boolean fetch);
47  private native boolean getOutputPixelsFetching(long handle);
48  private native void setBackgroundBitmap(long handle, Bitmap bitmap);
49  private native Scene.Layer pickLayer(long handle, float x, float y, boolean inPixels);

51  /**
52  * Scene coordinates to output (screen or bitmap) pixel coordinates mapping
53  */
54  public enum OutputMapping {
55  STRETCH, //!< output viewport covers entirely the scene axis span, aspect ratio is not preserved in general
56  FIT_WIDTH_TO_TOP, //!< width is covered entirely, height is resized to keep aspect ratio, the top borders are aligned
57  FIT_WIDTH, //!< width is covered entirely, height is resized to keep aspect ratio, point (0.5, 0.5) is mapped to the output center
58  FIT_HEIGHT //!< height is covered entirely, width is resized to keep aspect ratio, point (0.5, 0.5) is mapped to the output center
59  }

62  /**
63  * Creates new scene renderer
64  * @param context Beatmup core context
65  */
66  public SceneRenderer(Context context) {
67  super(context, newSceneRenderer(context));
68  }
69 
70  /**
71  * Attaches a bitmap to the renderer output.
72  * This disables the on-screen rendering.
73  * @param bitmap the bitmap the scene will be rendered to
74  */
75  public void setOutput(Bitmap bitmap) {
76  setOutput(handle, bitmap);
77  this.output = bitmap;
78  }
79 
80  /**
81  * Removes bitmap from the renderer output and switches to on-screen rendering.
82  * The rendering is done on the display currently connected to the Context running the rendering task.
83  */
84  public void resetOutput() {
85  this.output = null;
86  resetOutput(handle);
87  }
88 
89  /**
90  * @return a bitmap bound to the renderer or `null`.
91  */
92  public Bitmap getOutput() {
93  return output;
94  }
95 
96  /**
97  * Sets a scene to be rendered
98  * @param scene the scene
99  */
100  public void setScene(Scene scene) {
101  setScene(handle, scene);
102  this.scene = scene;
103  }
104 
105  /**
106  * @return a scene used by the renderer or null.
107  */
108  public Scene getScene() {
109  return this.scene;
110  }
111 
112  /**
113  * Sets the output mapping specifying how the scene coordinates [0,1]² are mapped to the output (screen or bitmap) pixel coordinates.
114  * @param mapping the new mapping
115  */
116  public void setOutputMapping(OutputMapping mapping) {
117  setOutputMapping(handle, mapping.ordinal());
118  }
119 
120  /**
121  * @return the output mapping specifying how the scene coordinates [0,1]² are mapped to the output (screen or bitmap) pixel coordinates.
122  */
123  public OutputMapping getOutputMapping() {
124  return OutputMapping.values()[ getOutputMapping(handle) ];
125  }
126 
127  /**
128  * Sets a value overriding output width for elements that have their size in pixels, in order to render a resolution-independent picture.
129  * If set negative or zero, the actual output width is taken.
130  * @param newWidth the new reference width value
131  */
132  public void setOutputReferenceWidth(int newWidth) {
133  setOutputReferenceWidth(handle, newWidth);
134  }
135 
136  /**
137  * @return Retrieves value overriding output width for elements that have their size in pixels, in order to render a resolution-independent picture
138  */
139  public int getOutputReferenceWidth() {
140  return getOutputReferenceWidth(handle);
141  }
142 
143  /**
144  * Sets an image to pave the background.
145  * @param bitmap the image
146  */
147  public void setBackground(Bitmap bitmap) {
148  this.background = bitmap;
149  if (bitmap != null)
150  setBackgroundBitmap(handle, bitmap);
151  else
152  setBackgroundBitmap(handle, null);
153  }
154 
155  /**
156  * @return the bitmap currently set to fill the background, null if not set.
157  */
158  public Bitmap getBackground() {
159  return background;
160  }
161 
162  /**
163  * Specifies whether the output image data is pulled from GPU to CPU memory every time the rendering is done.
164  * This is convenient if the rendered image is an application output result, and is further stored or sent through the network.
165  * Otherwise, if the image is to be further processed inside %Beatmup, the pixel transfer likely introduces an unnecessary latency and may cause
166  * FPS drop in real-time rendering.
167  * Has no effect in on-screen rendering.
168  * @param fetch If `true`, pixels are pulled to CPU memory.
169  */
170  public void setOutputPixelsFetching(boolean fetch) {
171  setOutputPixelsFetching(handle, fetch);
172  }
173 
174  public boolean getOutputPixelFetching() {
175  return getOutputPixelsFetching(handle);
176  }
177 
178  /**
179  * Renders the scene
180  * @return rendering time in ms
181  */
182  public float render() throws CoreException {
183  return context.performTask(this);
184  }
185 
186  /**
187  * Retrieves a scene layer visible at a given point, if any.
188  * In contrast to Scene::getLayer() it takes into account the output mapping.
189  * "Phantom" layers are ignored.
190  * @param x X coordinate.
191  * @param y Y coordinate.
192  * @param inPixels If `true`, the coordinates taken in pixels.
193  * @return the topmost layer at the given point if any, null otherwise.
194  */
195  public Scene.Layer pickLayer(float x, float y, boolean inPixels) {
196  return pickLayer(handle, x, y, inPixels);
197  }
