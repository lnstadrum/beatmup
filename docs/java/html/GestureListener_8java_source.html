50  public GestureListener(float pointerTolerance) {
51  this.squaredPointerTolerance = pointerTolerance * pointerTolerance;
52  cumulatedGesture = new AffineMapping();
53  lastMovement = new AffineMapping();
54  currentGesture = new AffineMapping();
55  tapping = false;
56  minGestureScale = 0.0f;
57  maxGestureScale = Float.POSITIVE_INFINITY;
58  animator = null;
59  holdEventTicket = Animator.INVALID_TICKET;
60  holdEvent = new Animator.Event() {
61  @Override
62  public long run() {
63  if (tapping)
64  onHold(mainX, mainY);
65  return -1;
66  }
67  };
68  }

72  if (animator != null)
73  animator.cancel(holdEvent, holdEventTicket);
74  }

78  cumulatedGesture.assign(currentGesture);

83  /**
84  * Retrieves a mapping from two pointers displacement. The default implementation assumes
85  * orthogonal mappings in image plane.
86  * @param target the retrieved mapping (output)
87  * @param X1 first pointer initial horizontal coordinate
88  * @param Y1 first pointer initial vertical coordinate
89  * @param X2 second pointer initial horizontal coordinate
90  * @param Y2 second pointer initial vertical coordinate
91  * @param newX1 first pointer final horizontal coordinate
92  * @param newY1 first pointer final vertical coordinate
93  * @param newX2 second pointer final horizontal coordinate
94  * @param newY2 second pointer final vertical coordinate
95  */
96  protected void resolveGesture(
97  AffineMapping target,
98  float X1, float Y1,
99  float X2, float Y2,
100  float newX1, float newY1,
101  float newX2, float newY2
102  ) {
103  if (Float.isNaN(X2) || Float.isNaN(Y2))
104  // convention: if point 2 shows NaNs, there is only one pointer given
105  target.setTranslation(newX1 - X1, newY1 - Y1);
106  else {
107  float scale = (float) Math.sqrt(cumulatedGesture.det());
108  target.resolve(
109  X1, Y1, X2, Y2,
110  newX1, newY1, newX2, newY2,
111  minGestureScale / scale, maxGestureScale / scale
112  );
113  }
114  }

117  /**
118  * To be called when a two-pointer gesture is performed
119  * @param X1 first pointer initial horizontal coordinate
120  * @param Y1 first pointer initial vertical coordinate
121  * @param X2 second pointer initial horizontal coordinate
122  * @param Y2 second pointer initial vertical coordinate
123  * @param newX1 first pointer final horizontal coordinate
124  * @param newY1 first pointer final vertical coordinate
125  * @param newX2 second pointer final horizontal coordinate
126  * @param newY2 second pointer final vertical coordinate
127  */
128  public void gestureUpdated(
129  float X1, float Y1,
130  float X2, float Y2,
131  float newX1, float newY1,
132  float newX2, float newY2
133  ) {
134  resolveGesture(lastMovement,
135  X1, Y1, X2, Y2,
136  newX1, newY1, newX2, newY2
137  );
138 
139  // if tapping, check main pointer displacement
140  if (tapping) {
141  float dx = Math.abs(mainX - newX1), dy = Math.abs(mainY - newY1);
142  if (dx*dx + dy*dy > squaredPointerTolerance) {
143  // stop tapping
144  tapping = false;
145  cancelPlannedEvents();
146  }
147  }
148 
149  AffineMapping.compose(currentGesture, lastMovement, cumulatedGesture);
150  if (!onGesture(currentGesture))
151  AffineMapping.compose(lastMovement, currentGesture, cumulatedGesture.getInverse());
152  }

155  public final void cancelGesture() {
156  if (pointerCount > 0)
157  onCancel();
158  pointerCount = 0;
159  tapping = false;
160  }

163  public final void pointerDown(float x, float y) {
164  cancelPlannedEvents();
165  if (pointerCount == 0) {
166  mainTouchTime = System.currentTimeMillis();
167  onFirstTouch(x,y);
168  mainX = x;
169  mainY = y;
170  tapping = true;
171  // planify hold event
172  if (animator != null) {
173  holdEventTicket = animator.planify(holdEvent, HOLD_TIME_MILLISEC);
174  }
175  } else
176  tapping = false;
177  onPointerDown(pointerCount, x, y);
178  pointerCount++;
179  }
180 
181 
182  public final void pointerUp() {
183  cancelPlannedEvents();
184  if (pointerCount == 1 && System.currentTimeMillis() < mainTouchTime + TAP_TIME_MILLISEC && tapping)
185  onTap(mainX, mainY);
186  else
187  if (pointerCount == 1)
188  onRelease(currentGesture);
189  pointerCount = Math.max(pointerCount-1, 0);
190  }
