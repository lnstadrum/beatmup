32 /**
33  Flood fill algorithm implementation.
34  Discovers areas of similar colors up to a tolerance threshold around given positions (seeds) in the input image. These areas are filled with white color in another image (output).
35  If the output bitmap is a binary mask, corresponding pixels are set to `1`. The rest of the output image remains unchanged.
36  Optionally, computes contours around the discovered areas and stores the contour positions. Also optionally, applies post-processing by dilating or eroding the discovered regions
37  in the output image.
38  */
39 public class FloodFill extends Task {

116  * @param releaseRadius Erosion/dilation radius of transition from 1 to 0
117  */
118  public void setBorderPostprocessing(BorderMorphology operation, float holdRadius, float releaseRadius) {
119  if (holdRadius < 0)
120  throw new IllegalArgumentException("Hold radius must not be negative.");
121  if (releaseRadius < holdRadius)
122  throw new IllegalArgumentException("Release radius must be greater than hold radius.");
123  setBorderPostprocessing(handle, operation.ordinal(), holdRadius, releaseRadius);

126  /**
127  * Communicates a set of seed points.
128  * The research of connected components of similar colors starts in specific locations in the input image (seeds)
129  * @param seeds The list of seeds
130  */
131  public void setSeeds(List<IntPoint> seeds) {
132  if (seeds.size() == 0)
133  throw new IllegalArgumentException("Seed set is empty.");
134  int[] xy = new int[seeds.size()*2];
135  for (int i = 0; i < seeds.size(); i++) {
136  xy[2*i ] = seeds.get(i).x;
137  xy[2*i+1] = seeds.get(i).y;
138  }
139  setSeeds(handle, xy);
140  }

197  /**
198  * Crops zero parts of the output mask.
199  * @param normalizedCropRect The crop rectangle in normalized coordinates (if not null, will be set)
200  * @return a new bitmap containing the cropped mask
201  */
202  public Bitmap optimizeMask(Rectangle normalizedCropRect) {
203  if (output == null)
204  throw new NullPointer("Output bitmap is not set.");
205  IntRectangle boundingBox = getBoundingBox(handle);
206  boundingBox.x2++;
207  boundingBox.y2++;
208  Bitmap result = output.copyRegion(boundingBox);
209  if (normalizedCropRect != null) {
210  normalizedCropRect.x1 = (float) boundingBox.x1 / output.getWidth();
211  normalizedCropRect.y1 = (float) boundingBox.y1 / output.getWidth();
212  normalizedCropRect.x2 = (float) (boundingBox.x1 + result.getWidth()) / output.getWidth();
213  normalizedCropRect.y2 = (float) (boundingBox.y1 + result.getHeight()) / output.getWidth();
214  }
215  return result;
216  }
