25 /**
26  * Resamples an image to a given resolution.
27  * Implements different resampling approaches, including standard ones (bilinear, bicubic, etc.) and a neural network-based 2x upsampling approach dubbed as "x2".
28  */
29 public class Resampler extends Task {
30  /**
31  * Resampling mode (algorithm) specification.
32  */
33  public enum Mode {
34  NEAREST_NEIGHBOR, //!< zero-order: usual nearest neighbor
35  BOX, //!< "0.5-order": anti-aliasing box filter; identical to nearest neighbor when upsampling
36  LINEAR, //!< first order: bilinear interpolation
37  CUBIC, //!< third order: bicubic interpolation
38  CONVNET //!< upsampling x2 using a convolutional neural network
39  };

47  private native float getCubicParameter(long handle);

49  /**
50  * Creates a resampler.
