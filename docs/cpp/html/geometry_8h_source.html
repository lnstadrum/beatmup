417  void rotateRadians(float angle) {
418  float
419  c = cos(angle),
420  s = sin(angle),
421  _11 = c * a11 + s * a12,
422  _21 = c * a21 + s * a22;
423  a12 = -s * a11 + c * a12;
424  a22 = -s * a21 + c * a22;
425  a11 = _11;
426  a21 = _21;

433  inline void skewRadians(float x, float y) {
434  float
435  tx = tan(x), ty = tan(y),
436  _11 = a11 * (1 + tx*ty) + a12 * ty,
437  _21 = a21 * (1 + tx*ty) + a22 * ty,
438  _12 = a11 * tx + a12;

440  a11 = _11;

442  a21 = _21;
