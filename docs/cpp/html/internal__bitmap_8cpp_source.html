41 InternalBitmap::InternalBitmap(Context& ctx, const char* bmpFilename) :
42  AbstractBitmap(ctx)
43 {
44  // read header
45  BmpFile bmp(bmpFilename);
46  switch (bmp.getBitsPerPixel()) {
47  case 1:
48  this->pixelFormat = PixelFormat::BinaryMask;
49  break;
50  case 4:
51  this->pixelFormat = PixelFormat::HexMask;
52  break;
53  case 8:
54  this->pixelFormat = PixelFormat::SingleByte;
55  break;
56  case 24:
57  this->pixelFormat = PixelFormat::TripleByte;
58  break;
59  case 32:
60  this->pixelFormat = PixelFormat::QuadByte;
61  break;
62  default:
63  throw IOError(bmpFilename, "Unsupported pixel format");
64  }
65  this->width = bmp.getWidth();
66  this->height = bmp.getHeight();
67 
68  // allocate & read
69  memory = AlignedMemory(getMemorySize());
70  bmp.load(memory(), getMemorySize());
71 }
